CREATE TABLE TEAM(
NAME varchar(20),
CITY varchar(20),
COACH varchar(20) NOT NULL,
CPID int,
PRIMARY KEY(NAME)	
);

CREATE TABLE PLAYER(
PID int,
NAME varchar(20),
TNAME varchar(20),
SKILLLEVEL int NOT NULL,
PRIMARY KEY(PID)
);

CREATE TABLE INJURY_RECORD(
PID int,
IDATE DATE,
DESCRIPTION varchar(20),
SEVER_LEVEl varchar(20),
PRIMARY KEY (DESCRIPTION,PID),
FOREIGN KEY(PID) REFERENCES PLAYER(PID)
);

CREATE TABLE GAME(
HOST_TEAM varchar(20),
GUEST_TEAM varchar(20),
GDATE DATE NOT NULL,
SCORE int NOT NULL,
PRIMARY KEY(HOST_TEAM,GUEST_TEAM),
FOREIGN KEY(HOST_TEAM) REFERENCES TEAM(NAME) ON DELETE SET NULL,
FOREIGN KEY(GUEST_TEAM) REFERENCES TEAM(NAME) ON DELETE SET NULL
);
